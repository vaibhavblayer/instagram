\title{speed of a point on circumference}

\vspace*{\fill}

\begin{center}
\begin{tikzpicture}[>=latex, scale=1.2]
\pic (ground) {frame=3.5cm};
\draw[very thick] (ground-center)++(0, 1) coordinate (o) circle[radius=1];
\draw[ -latex, rotate around= {120:(o)}] (o)++(0, 1) node[left]{$p$} --++(1, 0) ;
\draw[dashed, rotate around= {120:(o)}] (o)--++(0, 1);
\draw[dashed]  (o) node[right]{$c$}--++(0, -1);
\centerarc[thick] (o)(210:270:0.25);
\node at (-0.2, 0.65){$\theta$};
\begin{scope}[xshift=3cm, very thick]
	\draw[->] (0, 0) --(2, 0)node[right]{$v_c$};
	\draw[->, rotate around= {120:(0, 0)}] (0, 0) --(2, 0)node[left]{$v_{pc}$} ;
	\draw[->, rotate around= {60:(0, 0)}] (0, 0) --(2, 0) node[right]{$v_p$};
	\centerarc [thick] (0,0)(0:120:0.3);
	\node at (0.7,0.3){$180^\circ - \theta$};
\end{scope}
\end{tikzpicture}
\end{center}

{\Large{
\begin{align*}
\vec{v}_p &= \vec{v}_c + \vec{v}_{pc} \\[4mm]
 v_p &=2v\sin\dfrac{\theta}{2}
\end{align*}
}}
